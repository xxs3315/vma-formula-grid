<script lang="ts">
import { defineComponent, ref } from 'vue';

  export default defineComponent({
    name: 'Button',
    setup() {
      const isLoading = ref<boolean>(false);

      const clickEvent = () => {
        isLoading.value = !isLoading.value
      };

      return {
        isLoading,
        clickEvent,
      }
    },
  })
</script>
